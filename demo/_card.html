<!-- Simple Card -->
<div class="ny-card">
	<div class="ny-card-title">Title</div> <!-- only for title card -->
	<div class="ny-card-content">
		<!-- add content here -->
	</div>
</div>

<!-- Card with Action bar -->
<div class="ny-card">
	<div class="ny-card-title">Card Header</div> <!-- only for title card -->
	<div class="ny-card-content">
		<!-- add content here -->
	</div>
	<div class="ny-card-action">
		<a href="#" class="ny-button">Learn More</a> <!-- add button or link here -->
	</div>
</div>

<!-- Card with Header and Picture -->
<div class="ny-card">
	<div class="ny-card-header ny-flex--row"> <!-- header card with ny-flex--row for horizontal layout -->
		<img src="img/avatar.png" class="ny-avatar" alt=""> <!-- avatar or image profile -->
		<div class="ny-flex--col"> <!-- flex column for vertical layout -->
			<span class="ny-card-header-title">Title</span>
			<span class="ny-card-header-subtitle">subtitle</span>
		</div>
	</div>
	<div class="ny-card-content">
		<!-- add content here -->
	</div>
	<div class="ny-card-image"> <!-- Image Card -->
		<img src="foo/bar.png" alt=""> 
	</div>
	<div class="ny-card-action">
		<a href="#" class="ny-button">Share</a>
	</div>
</div>
